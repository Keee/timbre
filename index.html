Overview

Timbre.js is so easy. 

// generate a 523.25Hz(C4) sine
T("sin", 523.25).play();

It's useful and powerful. 

// generate a chord C-major            
T("+", T("sin", 523.35),
       T("sin", 659.25),
       T("sin", 783.99)).set({mul: 0.25}).play();

Looks a bit like CUI Max/MSP??? SuperCollider?? 

// tremolo & decay            
T("*", T("+", T("sin", 523.35),
              T("sin", 659.25),
              T("sin", 783.99)).set({mul: 0.25}),
       T("+tri", 2),
       T("adsr", 100, 2500)).play();

You can so easily learn it. Because it is pure JavaScript. Fun!!! 

timbre.utils.exports("atof");

var synth = T("efx.delay", 150, 0.8);

function tone(freq) {
    synth.append( T("*", T("+", T("sin", freq, 0.25),
                                T("sin", freq + 4, 0.1)),
                         T("perc", 2500).bang()) );
    if (synth.args.length > 4) synth.args.shift();
}

var sch = T("schedule", "bpm (116, 2)", [
    [0, tone, [atof("G3")]],
    [1, tone, [atof("B4")]], [1, tone, [atof("D5")]], [1, tone, [atof("F#5")]],
    [3, tone, [atof("D3")]],
    [4, tone, [atof("A4")]], [4, tone, [atof("C#5")]], [4, tone, [atof("F#5")]],
    [6],
], true);

synth.onplay = function() {
    sch.on();
};
synth.onpause = function() {
    sch.off();
};
synth.play();
